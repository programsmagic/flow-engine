Super Simple Usage

# Install Flow globally
npm install -g flow

# Generate workflows for your project
flow generate

# Analyze without generating files
flow analyze

# Detect frameworks and languages
flow detect

# Get optimization suggestions
flow optimize

Programmatic Usage

import { Flow } from 'flow';

const flow = new Flow({
  framework: 'react',
  parallel: true,
  cache: true
});

// Generate workflows
const result = await flow.generate({
  input: './src',
  output: './workflows'
});

// Execute workflows asynchronously
const execution = await flow.executeWorkflow('user-authentication', {
  userId: 123,
  action: 'login'
});

// Get performance metrics
const metrics = flow.getMetrics();
console.log('Success rate:', metrics.successRate);
